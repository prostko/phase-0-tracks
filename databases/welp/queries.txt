CREATE TABLE businesses (
id INTEGER PRIMARY KEY,
name VARCHAR(255)
);

CREATE TABLE users (
id INTEGER PRIMARY KEY,
first_name VARCHAR(255),
last_name VARCHAR(255)
);

CREATE TABLE reviews (
id INTEGER PRIMARY KEY,
stars INTEGER,
comment VARCHAR(255),
business_id INTEGER,
user_id INTEGER,
FOREIGN KEY (user_id) REFERENCES users(id),
FOREIGN KEY (business_id) REFERENCES businesses(id)
 );
